Apollo动态修改log4j2.xml

三万 · 2021-7-28 · 次阅读

log4j2

首先resouse目录下创建log4j2.xml
application里面配置 logging.config = classpath:log4j2/log4j2.xml

            <?xml version="1.0" encoding="UTF-8" ?>
            <Configuration status="INFO" monitorInterval="30">
                <Appenders>
                    <Console name="console" target="SYSTEM_OUT">
                        <PatternLayout pattern="%d{yyyy-MM-dd HH:mm:ss.SSS} [%t] [%-5p] {%F:%L} - %m%n" />
                    </Console>
                    <!--使用redis启用下面配置-->
                    <RedisAppender name="redisAppender" appName="log" redisHost="${ctx:redisurl}" >
                        <PatternLayout pattern="%d{yyyy-MM-dd HH:mm:ss.SSS} [%t] [%-5p] {%F:%L} - %m%n" />
                    </RedisAppender>
                </Appenders>
                <Loggers>
                    <root level="info">
                        <appender-ref ref="redisAppender"/>
                        <appender-ref ref="console"/>
                    </root>
                </Loggers>
            </Configuration>

自定义ApolloConfig

package com.apollolog.demo.config;

import com.ctrip.framework.apollo.Config;
import com.ctrip.framework.apollo.ConfigChangeListener;
import com.ctrip.framework.apollo.ConfigService;
import com.ctrip.framework.apollo.model.ConfigChange;
import com.ctrip.framework.apollo.model.ConfigChangeEvent;
import lombok.extern.slf4j.Slf4j;

@Slf4j
public class ApolloConfig {
    private static ApolloConfig instance;
    private static Object locked = new Object();
    private Config config;
    private String DEFAULT_VALUE = "undefined";

    private ApolloConfig()
    {
        ConfigChangeListener changeListener = new ConfigChangeListener() {
            @Override
            public void onChange(ConfigChangeEvent changeEvent) {
                log.info("Changes for namespace {}", changeEvent.getNamespace());
                for (String key : changeEvent.changedKeys()) {
                    ConfigChange change = changeEvent.getChange(key);
                    log.info("Change - key: {}, oldValue: {}, newValue: {}, changeType: {}",
                            change.getPropertyName(), change.getOldValue(), change.getNewValue(),
                            change.getChangeType());
                }
            }
        };
        config = ConfigService.getAppConfig();
        config.addChangeListener(changeListener);
    }
    private ApolloConfig(String namespace)
    {
        ConfigChangeListener changeListener = new ConfigChangeListener() {
            @Override
            public void onChange(ConfigChangeEvent changeEvent) {
                log.info("Changes for namespace {}", changeEvent.getNamespace());
                for (String key : changeEvent.changedKeys()) {
                    ConfigChange change = changeEvent.getChange(key);
                    log.info("Change - key: {}, oldValue: {}, newValue: {}, changeType: {}",
                            change.getPropertyName(), change.getOldValue(), change.getNewValue(),
                            change.getChangeType());
                }
            }
        };
        config = ConfigService.getConfig(namespace);
        config.addChangeListener(changeListener);
    }

    public static ApolloConfig getInstance()
    {
        if(instance ==null)
        {
            synchronized (locked) {
                if (instance == null) {
                    instance = new ApolloConfig();
                }
            }
        }
        return instance;
    }
    public static ApolloConfig getInstance(String namespace)
    {
        if(instance ==null)
        {
            synchronized (locked) {
                if (instance == null) {
                    instance = new ApolloConfig(namespace);
                }
            }
        }
        return instance;
    }
    public String getConfig(String key) {
        String result = config.getProperty(key, DEFAULT_VALUE);
        log.info(String.format("Loading key : %s with value: %s", key, result));
        return result;
    }

}

监听器

这里要注意ApolloConfig.getInstance(“log”).getConfig(“redisurl”)，修改为自己需要的namespace和要取的参数

package com.apollolog.demo.config;
import com.apollolog.demo.config.ApolloConfig;
import org.slf4j.MDC;
import org.springframework.boot.SpringApplication;
import org.springframework.boot.context.event.ApplicationEnvironmentPreparedEvent;
import org.springframework.boot.context.event.ApplicationFailedEvent;
import org.springframework.boot.context.event.ApplicationPreparedEvent;
import org.springframework.boot.context.event.ApplicationStartingEvent;
import org.springframework.context.ApplicationContext;
import org.springframework.context.ApplicationEvent;
import org.springframework.context.event.ContextClosedEvent;
import org.springframework.context.event.GenericApplicationListener;
import org.springframework.core.Ordered;
import org.springframework.core.ResolvableType;

public class ApplicationStartedEventListener implements GenericApplicationListener {


    public static final int DEFAULT_ORDER = Ordered.HIGHEST_PRECEDENCE + 10;

    private static Class<?>[] EVENT_TYPES = { ApplicationStartingEvent.class, ApplicationEnvironmentPreparedEvent.class,
            ApplicationPreparedEvent.class, ContextClosedEvent.class, ApplicationFailedEvent.class };

    private static Class<?>[] SOURCE_TYPES = { SpringApplication.class, ApplicationContext.class };

    @Override
    public void onApplicationEvent(ApplicationEvent event) {

        if (event instanceof ApplicationEnvironmentPreparedEvent) {
            String redisurl = ApolloConfig.getInstance("log").getConfig("redisurl");
            MDC.put("redisurl", redisurl);
            System.err.println("=========redishost" + redisurl);
        }

    }

    @Override
    public int getOrder() {
        return DEFAULT_ORDER;
    }

    @Override
    public boolean supportsEventType(ResolvableType resolvableType) {
        return isAssignableFrom(resolvableType.getRawClass(), EVENT_TYPES);
    }

    @Override
    public boolean supportsSourceType(Class<?> sourceType) {
        return isAssignableFrom(sourceType, SOURCE_TYPES);
    }

    private boolean isAssignableFrom(Class<?> type, Class<?>... supportedTypes) {
        if (type != null) {
            for (Class<?> supportedType : supportedTypes) {
                if (supportedType.isAssignableFrom(type)) {
                    return true;
                }
            }
        }
        return false;
    }
}

启动类

package com.apollolog.demo;

import com.apollolog.demo.config.ApplicationStartedEventListener;
import com.ctrip.framework.apollo.spring.annotation.EnableApolloConfig;
import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;

@EnableApolloConfig
@SpringBootApplication
public class DemoApplication {

    public static void main(String[] args) {

        SpringApplication app = new SpringApplication(DemoApplication.class);

        ApplicationStartedEventListener asel = new ApplicationStartedEventListener();

        app.addListeners(asel);

        app.run(args);
    }

}
